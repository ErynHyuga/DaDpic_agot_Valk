mz_amazon_culture_fix_decision = {
	picture = "gfx/interface/illustrations/decisions/mz_decision_recruit_amazon.dds"
	major = yes
	ai_check_interval = 0
	ai_potential = {
		always = no
	}
	ai_will_do = {
		base = 0
	}

	is_shown = {

		OR = {

			AND = {
				global_var:mz_amazon_kingdom_founded ?= flag:mz_scythia
				culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_war_riders }
				NOR = { 
					culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazonian_enclaves } 
					culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_warriors } 
				}
			}

			AND = {
				global_var:mz_amazon_kingdom_founded ?= flag:mz_themiscyra
				OR = { 
					culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazonian_enclaves } 
					culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_warriors } 
				}
				NOT = { culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_mountaineers } }
			}

		}

	}

	effect = {

		if = {
			limit = {
				global_var:mz_amazon_kingdom_founded ?= flag:mz_scythia
				culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_war_riders }
				NOR = { 
					culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazonian_enclaves } 
					culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_warriors } 
				}
			}
			if = {
				limit = {
					culture:mz_amazonian = { has_cultural_tradition = tradition_swords_for_hire }
				}
				culture:mz_amazonian = { remove_culture_tradition = tradition_swords_for_hire }
			}
			if = {
				limit = {
					culture:mz_amazonian = { has_cultural_tradition = tradition_forbearing }
				}
				culture:mz_amazonian = { remove_culture_tradition = tradition_forbearing }
			}
			if = {
				limit = {
					culture:mz_amazonian = { has_cultural_tradition = tradition_collective_lands }
				}
				culture:mz_amazonian = { remove_culture_tradition = tradition_collective_lands }
			}
			every_culture_global = {
				limit = { 
					has_cultural_tradition = mz_tradition_amazon_war_riders 
					NOR = { 
						culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazonian_enclaves } 
						culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_warriors } 
					}
					OR = {
						this = culture:mz_amazonian
						any_parent_culture_or_above = { this = culture:mz_amazonian }
					}
				}
				add_culture_tradition = mz_tradition_amazon_warriors
			}
		}

		if = {
			limit = {
				global_var:mz_amazon_kingdom_founded ?= flag:mz_themiscyra
				OR = { 
					culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazonian_enclaves } 
					culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_warriors } 
				}
				NOT = { culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_mountaineers } }
			}
			if = {
				limit = {
					culture:mz_amazonian = { has_cultural_tradition = tradition_swords_for_hire }
				}
				culture:mz_amazonian = { remove_culture_tradition = tradition_swords_for_hire }
			}
			if = {
				limit = {
					culture:mz_amazonian = { has_cultural_tradition = tradition_forbearing }
				}
				culture:mz_amazonian = { remove_culture_tradition = tradition_forbearing }
			}
			if = {
				limit = {
					culture:mz_amazonian = { has_cultural_tradition = tradition_collective_lands }
				}
				culture:mz_amazonian = { remove_culture_tradition = tradition_collective_lands }
			}
			every_culture_global = {
				limit = { 
					OR = {
						this = culture:mz_amazonian
						any_parent_culture_or_above = { this = culture:mz_amazonian }
					}
					OR = { 
						culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazonian_enclaves } 
						culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_warriors } 
					}
					NOT = { culture:mz_amazonian = { has_cultural_tradition = mz_tradition_amazon_mountaineers } }
				}
				add_culture_tradition = mz_tradition_amazon_mountaineers
			}
		}		

	}
}
