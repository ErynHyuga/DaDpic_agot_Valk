mz_doctrine_lesbian_category = {
	group = "marriage"
	is_available_on_create = {
		OR = {
			exists = var:mz_lesbian_revelation_variable # Activated by lesbian revelation...
			has_doctrine_parameter = mz_lesbian_marriage_parameter # ...or already having a lesbian marriage doctrine.
		}
	}

	mz_doctrine_no_lesbian_doctrine = { # None. Will remove the category entirely upon creation (via on_action)
		piety_cost = {
			if = {
				limit = { 
					has_doctrine_parameter = mz_lesbian_marriage_parameter
				}
				value = faith_tenet_cost_massive	
			}
			else = { value = 0 }
		}
	}

	mz_doctrine_lesbian_marriage = { # Allows for lesbian marriage, but no other effects. Fewer requirements than other doctrines.
		can_pick = {
			custom_description = {
				text = mz_cannot_be_male_dominated_trigger
				NOT = { doctrine:doctrine_gender_male_dominated = { is_in_list = selected_doctrines } }
			}
		}

		piety_cost = {
			if = {
				limit = {
					OR = {
						has_doctrine = doctrine_homosexuality_accepted
						has_doctrine = mz_doctrine_homosexuality_embraced
						has_doctrine = mz_doctrine_lesbian_divinity
						has_doctrine = mz_doctrine_sapphic_exaltation
					}
				}
				value = faith_tenet_cost_low				
			}
			else_if = {
				limit = {
					OR = {
						religion_tag = buddhism_religion
						religion_tag = jainism_religion
						religion_tag = hinduism_religion
					}
				}
				value = faith_tenet_cost_high
			}
			else_if = {
				limit = {
					has_doctrine = abrahamic_hostility_doctrine
				}
				value = faith_tenet_cost_massive
			}
			else = {
				value = faith_tenet_cost_mid
			}

			# Multiplier for keeping same tenet
			if = {
				limit = { has_doctrine = mz_doctrine_lesbian_marriage }
				multiply = faith_unchanged_doctrine_cost_mult
			}
		}
	
		parameters = {
			mz_lesbian_marriage_parameter = yes	# Enables lesbian marriage.
			mz_lesbian_consorts_parameter = yes # Piety bonus for female consorts.
		}
	}
