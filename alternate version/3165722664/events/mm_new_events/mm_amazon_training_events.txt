namespace = mz_amazon_training

mz_amazon_training.0001 = { # Null event, each time it triggers it has a reduced chance of being repeated
	hidden = yes

	weight_multiplier = {
		base = 1
		modifier = {
			exists = var:mz_amazon_training_null
			add = {
				value = var:mz_amazon_training_null
				multiply = -0.25
			}
		}
	}

	immediate = {
		if = {
			limit = { exists = var:mz_amazon_training_null }
			change_variable = {
				name = mz_amazon_training_null
				add = 1
			}
		}
		else = {
			set_variable = {
				name = mz_amazon_training_null
				value = 1
				years = 10
			}
		}
	}
}

mz_amazon_training.1001 = { # Gain Prowess
	hidden = yes
	cooldown = { years = 1 }

	trigger = {
		any_courtier = { mz_amazon_trainee = yes }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			any_courtier = {
				count = all
				mz_amazon_trainee = yes
				prowess >= decent_skill_rating
			}
			add = -0.5
		}
		modifier = {
			any_courtier = {
				count = all
				mz_amazon_trainee = yes
				prowess >= high_skill_rating
			}
			factor = 0.1
		}
	}

	immediate = {
		random_courtier = {
			limit = { mz_amazon_trainee = yes }
			weight = {
				base = 1
				modifier = {
					add = 10
					is_of_major_interest_to_root_trigger = yes
				}
				modifier = {
					add = 5
					is_of_minor_interest_to_root_trigger = yes
				}
				modifier = {
					add = 5
					prowess < low_skill_rating
				}
				modifier = {
					add = 5
					prowess < mediocre_skill_rating
				}
				modifier = {
					factor = 0.1
					OR = {
						prowess >= high_skill_rating
						has_trait = mz_amazonian_warrior
					}
				}
			}
			save_scope_as = mz_amazon_trainee
		}
		if = {
			limit = { exists = scope:mz_amazon_trainee }
			scope:mz_amazon_trainee = { mz_amazon_training_progress_effect = yes }
			random_list = {
				1 = {
					send_interface_message = {
						type = event_marshal_task_good
						title = mz_amazon_training_event.t
						desc = mz_amazon_training_event.desc
						left_icon = scope:mz_amazon_trainee
						scope:mz_amazon_trainee = { add_prowess_skill = 2 }
					}
				}
				4 = {
					send_interface_message = {
						type = event_marshal_task_good
						title = mz_amazon_training_event.t
						desc = mz_amazon_training_event.desc
						left_icon = scope:mz_amazon_trainee
						scope:mz_amazon_trainee = { add_prowess_skill = 1 }
					}
				}
			}
			if = {
				limit = { 
					has_variable = mz_amazon_training_null
				}
				remove_variable = mz_amazon_training_null
			}
			scope:mz_amazon_trainee = {
				add_character_flag = {
					flag = mz_amazon_event_fired_recently
					years = 3
				}
			}
		}
	}
}
